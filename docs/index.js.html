const Client = require('./lib/Client');

/**
 * Creates a new instance of YADL.js.
 * @param {string} token The bot token.
 * If using a user account, the token is the one found in local storage.
 * @param {object} [options={}] Options for the client.
 * @example
 * new YADL('bot_token');
 * @example
 * new YADL('bot_token', {disableEvents: [Constants.Events.TYPING_START, Constants.Events.TYPING_STOP]})
 * @see {@link YADL#Event}
 */
function YADL(token, options = {}) {
	return new Client(token, options);
}

YADL.Constants = require('./lib/util/Constants');

module.exports = YADL;
